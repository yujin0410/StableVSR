\makeatletter
\def\@xfloat#1[#2]{%
  \@nodocument
  \def\@captype{#1}%
  \def\@fps{#2}%
  \@onelevel@sanitize\@fps
  \def\reserved@b{!}%
  \ifx\@fps\reserved@b
      \@fpsadddefault
  \else
      \ifx\@fps\@empty \@fpsadddefault \fi
  \fi
  \ifhmode
    \@bsphack
    \@floatpenalty -\@Mii
  \else
    \@floatpenalty-\@Miii
  \fi
  \ifinner
    \@parmoderr\@floatpenalty\z@
  \else
    \@next\@currbox\@freelist
        {\@tempcnta\sixt@@n
         \expandafter\@tfor \expandafter \reserved@a
                \expandafter :\expandafter =\@fps
            \do{\if \reserved@a h\advance\@tempcnta \@ne\fi
                \if \reserved@a t\advance\@tempcnta \tw@\fi
                \if \reserved@a b\advance\@tempcnta 4\relax\fi
                \if \reserved@a p\advance\@tempcnta 8\relax\fi
                \if \reserved@a !\@tempswatrue \fi
               }%
         \@tempcntb\csname ftype@#1\endcsname
         \multiply\@tempcntb\@xxxii
         \advance\@tempcnta\@tempcntb
         \global\count\@currbox\@tempcnta}%
        \@fltovf
  \fi
  \global\setbox\@currbox
        \color@vbox
        \normalcolor
        \vbox\bgroup
        \hsize\columnwidth
        \@parboxrestore
        \@floatboxreset
}
\makeatother
